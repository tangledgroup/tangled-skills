# scipy.io - Input/Output Operations

The `scipy.io` module provides tools for reading and writing various file formats commonly used in scientific computing, including MATLAB files, WAV audio files, ARFF data files, and NetCDF.

## MATLAB File I/O

### Reading MATLAB Files (.mat)

```python
from scipy import io
import numpy as np

# Read MATLAB file (v7.3 uses HDF5 format)
mat_data = io.loadmat('file.mat')

# mat_data is a dictionary with variable names as keys
print(mat_data.keys())  # Variable names in the .mat file

# Access variables
array_var = mat_data['variable_name']  # NumPy array
struct_var = mat_data['struct_name']    # Struct becomes nested dict

# MATLAB cell arrays become lists of arrays
cell_var = mat_data['cell_array']
```

### Writing MATLAB Files

```python
from scipy import io
import numpy as np

# Create data to save
data_dict = {
    'array1': np.random.rand(10, 20),
    'array2': np.array([1, 2, 3, 4, 5]),
    'string_var': 'Hello MATLAB'
}

# Save to MATLAB file (MATLAB 7.3 format, HDF5-based)
io.savemat('output.mat', data_dict)

# With specific format version
io.savemat('output.mat', data_dict, 
           format='octave')  # For GNU Octave compatibility
```

### Reading Specific Variables

```python
from scipy import io

# Load only specific variables (more memory efficient)
mat_data = io.loadmat('large_file.mat', 
                      squeeze_me=True,    # Remove singleton dimensions
                      mat_dtype=True)     # Preserve MATLAB data types

# Access variable
specific_var = mat_data['my_variable']
```

### Handling MATLAB Structures and Cells

```python
from scipy import io
import numpy as np

# MATLAB struct becomes dict of arrays
struct_data = io.loadmat('struct_file.mat')
field1 = struct_data['my_struct'][0][0]['field1']  # Nested access

# Cell arrays become tuples/arrays
cell_data = struct_data['cell_array'][0]  # First element (tuple)
element_0 = cell_data[0]  # First cell
```

## WAV Audio File I/O

### Reading WAV Files

```python
from scipy.io import wavfile
import numpy as np

# Read WAV file
sample_rate, data = wavfile.read('audio.wav')

print(f"Sample rate: {sample_rate} Hz")
print(f"Data shape: {data.shape}")  # (samples,) for mono, (samples, channels) for stereo
print(f"Data dtype: {data.dtype}")  # int16, int32, or float32

# For stereo audio
left_channel = data[:, 0]
right_channel = data[:, 1]

# Calculate duration
duration_seconds = len(data) / sample_rate
```

### Writing WAV Files

```python
from scipy.io import wavfile
import numpy as np

# Generate simple sine wave
sample_rate = 44100  # Standard CD quality
frequency = 440      # A4 note
duration = 1.0       # 1 second

t = np.linspace(0, duration, int(sample_rate * duration))
audio_data = 0.5 * np.sin(2 * np.pi * frequency * t)

# Convert to int16 (standard for WAV)
audio_int16 = (audio_data * 32767).astype(np.int16)

# Write WAV file
wavfile.write('output.wav', sample_rate, audio_int16)

# For stereo, shape data as (samples, 2)
stereo_data = np.column_stack([audio_int16, audio_int16])
wavfile.write('stereo_output.wav', sample_rate, stereo_data)
```

### Handling Different Audio Formats

```python
from scipy.io import wavfile
import numpy as np

# Read and convert to float for processing
sample_rate, data_int = wavfile.read('audio.wav')

# Convert int16 to float [-1, 1]
data_float = data_int.astype(np.float32) / 32768.0

# Process audio (example: normalize)
data_normalized = data_float / np.max(np.abs(data_float))

# Convert back to int16 for saving
data_out = (data_normalized * 32767).astype(np.int16)
wavfile.write('normalized.wav', sample_rate, data_out)
```

## ARFF File I/O (Weka Format)

### Reading ARFF Files

```python
from scipy.io import arff
import numpy as np

# Read ARFF file
data_dict = arff.loadarff('data.arff')

# Returns tuple: (data_array, attribute_info)
data, attributes = data_dict

print(f"Data shape: {data.shape}")
print(f"Attributes: {list(attributes.keys())}")

# Access specific attribute
attribute_values = data['attribute_name']

# Convert to NumPy array for numerical attributes
numeric_data = np.array([[row[attr] for attr in numeric_attrs] 
                         for row in data])
```

### Writing ARFF Files

```python
from scipy.io import arff
import numpy as np

# Create data in ARFF format
data = [
    {'age': 25, 'income': 50000.0, 'class': 'yes'},
    {'age': 35, 'income': 60000.0, 'class': 'no'},
    {'age': 45, 'income': 70000.0, 'class': 'yes'},
]

# Write to ARFF file
with open('output.arff', 'w') as f:
    arff.dumparff(data, f)

# With header (attribute definitions)
header = '''@relation example

@attribute age numeric
@attribute income numeric
@attribute class {yes, no}

@data
'''

with open('output.arff', 'w') as f:
    arff.dumparff(data, f, header=header)
```

### ARFF Data Types

```python
from scipy.io import arff

# ARFF supports these attribute types:
# - numeric: Real numbers
# - integer: Integers  
# - nominal: Categorical (enum-like)
# - string: Text strings
# - date: Date/time values

data_mixed = [
    {'numeric_attr': 3.14, 'nominal_attr': 'category_a', 
     'string_attr': 'text value'},
    {'numeric_attr': 2.71, 'nominal_attr': 'category_b',
     'string_attr': 'another text'},

```

## Practical Examples

### Processing Audio Signals

```python
from scipy.io import wavfile
from scipy import signal
import numpy as np

# Load audio file
sample_rate, audio_data = wavfile.read('speech.wav')

# Convert to mono if stereo
if len(audio_data.shape) > 1:
    audio_data = audio_data.mean(axis=1)

# Normalize to [-1, 1]
audio_float = audio_data.astype(np.float32) / np.iinfo(audio_data.dtype).max

# Apply filter (example: lowpass at 4kHz)
nyquist = sample_rate / 2
b, a = signal.butter(4, 4000/nyquist, btype='low')
audio_filtered = signal.filtfilt(b, a, audio_float)

# Save processed audio
audio_out = (audio_filtered * np.iinfo(np.int16).max).astype(np.int16)
wavfile.write('filtered_speech.wav', sample_rate, audio_out)
```

### Loading and Saving MATLAB Results

```python

import numpy as np

# 1. Export data from Python to MATLAB

```

### Working with Machine Learning Datasets

```python
from scipy.io import arff
import numpy as np

# Load UCI dataset in ARFF format
data_dict = arff.loadarff('iris.arff')
data, attributes = data_dict

# Extract features and target
feature_names = [attr for attr in attributes.keys() 
                 if attr != 'class']  # Assuming 'class' is target

X = np.array([[float(row[feat]) for feat in feature_names] 
              for row in data])
y = np.array([row['class'].decode() if isinstance(row['class'], bytes) 
              else row['class'] for row in data])

print(f"Features shape: {X.shape}")
print(f"Target shape: {y.shape}")
```

## Troubleshooting

### MATLAB Version Compatibility

```python
from scipy import io
import numpy as np

# For MATLAB 7.3 (HDF5) files, use loadmat directly
try:
    data = io.loadmat('file.mat')
except OSError as e:
    print(f"File format issue: {e}")
    # Try with different options
    data = io.loadmat('file.mat', 
                      matlab_compatible=True,
                      append_mat=True)

# Save in compatible format
io.savemat('output.mat', data_dict, 
           short_names=True)  # Truncate long variable names
```

### WAV File Format Issues

```python
from scipy.io import wavfile
import numpy as np

# Handle different bit depths
sample_rate, data = wavfile.read('audio.wav')

if data.dtype == np.int32:
    # Convert to int16
    data_converted = (data // 256).astype(np.int16)
elif data.dtype == np.float32:
    # Convert to int16
    data_converted = (data * 32767).astype(np.int16)

wavfile.write('converted.wav', sample_rate, data_converted)
```
